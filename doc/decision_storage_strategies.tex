% UML class diagram of strategy design pattern for using different knowledge repositories
\begin{tikzpicture}[font = \sffamily, thick]
\tikzumlset{font = \sffamily, fill class=white, fill package=white, fill note=white}

% abstract strategy/interface
\umlclass[type=interface]{IPersistenceStrategy}{}{
    \textit{+ insertDecisionKnowledgeElement(element : DecisionKnowledgeElement) : DecisionKnowledgeElement}\\
    \textit{+ updateDecisionKnowledgeElement(element : DecisionKnowledgeElement) : boolean}\\
    \textit{+ deleteDecisionKnowledgeElement(element : DecisionKnowledgeElement) : boolean}\\
    \textit{+ getDecisionKnowledgeElements(projectKey : String) : List<DecisionKnowledgeElement>}\\
    \textit{+ getDecisionKnowledgeElement(key : String) : DecisionKnowledgeElement}\\
    \textit{+ getDecisionKnowledgeElement(id : long) : DecisionKnowledgeElement}\\
    \textit{+ getDecisions(projectKey : String) : List<DecisionKnowledgeElement>}\\
    \textit{+ getChildren(element : DecisionKnowledgeElement) : List<DecisionKnowledgeElement>}\\
    \textit{+ getParents(element : DecisionKnowledgeElement) : List<DecisionKnowledgeElement>}\\
    \textit{+ getUnlinkedDecisionComponents(element : DecisionKnowledgeElement) : List<DecisionKnowledgeElement>}\\
    
    \textit{+ insertLink(link : Link) : long}\\
    \textit{+ deleteLink(link : Link) : boolean}\\
    \textit{+ getInwardLinks(element : decisionKnowledgeElement) : List<Link>}\\
    \textit{+ getOutwardLinks(element : decisionKnowledgeElement) : List<Link>}\\
}

% concrete strategy
\umlclass[x=12, y=1]{IssueStrategy}{}{}
\umlclass[x=12, y=-1]{ActiveObjectStrategy}{}{}

% client
\umlclass[x=-10.5, y=0]{Client}{}{}

% associations
\umlimpl{IssueStrategy}{IPersistenceStrategy}
\umlimpl{ActiveObjectStrategy}{IPersistenceStrategy}
\umluniassoc[arg1=uses, pos1=0.5]{Client}{IPersistenceStrategy}
\end{tikzpicture}
